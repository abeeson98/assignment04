Data Visualization 1:

Counting the number of insured individuals

options(scipen=999)
wdata %>%
  filter(AGE>18) %>%
  mutate(race_string=
    case_when(
      RACE ==1 ~ "White",
      RACE ==2 ~ "Black",
      RACE ==3 ~ "American Indian/Alaskan Native",
      RACE ==4 | 5 | 6 ~ "Asian/Pacific Islander",
      RACE == 7 | 8| 9 ~ "Other Race"
    )
  ) %>%
  group_by(race_string) %>%
  summarize(nins=
              sum(HCOVANY==2)) %>%
  ggplot() +
  geom_col(mapping=aes(x=race_string, y=nins)) +
  labs(
    x= "Race Category",
    y= "Count of Insured Individuals",
    title= "Counting the Number of People with Any Insurance",
    subtitle= "Comparing the numbers of people with insurance by race"
  )

Data Visualization 2:

Comparing the proportions of people who are insured

options(scipen=999)
wdata%>%
  mutate(anyins=
           case_when(HCOVANY==1 ~ 0, HCOVANY==2 ~ 1)) %>%
  filter(AGE>18) %>%
  mutate(race_string=
    case_when(
      RACE ==1 ~ "White",
      RACE ==2 ~ "Black",
      RACE ==3 ~ "American Indian/Alaskan Native",
      RACE ==4 | 5 | 6 ~ "Asian/Pacific Islander",
      RACE == 7 | 8| 9 ~ "Other Race"
    )
  ) %>%
  group_by(race_string) %>%
  summarize(propinsurance=
              mean(anyins)) %>%
  ggplot() +
  geom_col(mapping=aes(x=race_string, y=propinsurance)) +
  labs(
    x= "Race Category",
    y= "Proportion of Insured Individuals",
    title= "Understanding the Number of People with Any Insurance",
    subtitle= "Comparing the proportions of people with insurance by race"
  )
